// ===================== CONFIGURAÇÕES =====================
#define DEVICE_ID "printpixel"  // ALTERE AQUI O ID DO SEU TOTEM
#define SERVER_URL "https://totem-server.onrender.com"  // NÃO ALTERAR

#define RESET_BUTTON 0
#define BOTAO_COR 15
#define BOTAO_MAIS 4
#define BOTAO_MENOS 5
#define BOTAO_CORACAO 18
#define LED_PIN 2
#define LED_FITA_PIN 12
#define LED_CARDIO_PIN 13

#define MP3_RX_PIN 26
#define MP3_TX_PIN 27

#define NUM_LEDS 100
#define NUM_LEDS_CARDIO 15
#define BRILHO_MAX 255
#define BRILHO_MIN 50
#define EEPROM_SIZE 128

const uint32_t cores[] = {
  Adafruit_NeoPixel::Color(255, 255, 255),
  Adafruit_NeoPixel::Color(255, 255, 200),
  Adafruit_NeoPixel::Color(255, 255, 0),
  Adafruit_NeoPixel::Color(0, 0, 255),
  Adafruit_NeoPixel::Color(255, 0, 255),
  Adafruit_NeoPixel::Color(255, 0, 0),
  Adafruit_NeoPixel::Color(0, 255, 0),
  Adafruit_NeoPixel::Color(255, 165, 0),
  Adafruit_NeoPixel::Color(128, 0, 128),
  Adafruit_NeoPixel::Color(0, 255, 255),
  0,
  0
};
#define NUM_CORES 12

Adafruit_NeoPixel fitaPrincipal(NUM_LEDS, LED_FITA_PIN, NEO_GRB + NEO_KHZ800);
Adafruit_NeoPixel fitaCardio(NUM_LEDS_CARDIO, LED_CARDIO_PIN, NEO_GRB + NEO_KHZ800);
WiFiClient espClient;
PubSubClient client(espClient);
WiFiManager wm;
HardwareSerial mp3Serial(2);

int corAtual = 0;
int brilhoAtual = 255;
bool efeitoAtivo = false;
unsigned long efeitoInicio = 0;

unsigned long ledQrCodeTimer = 0;
bool ledQrCodeState = false;
bool ledQrCodeAtivo = false;
const int ledQrCodePiscadas = 5;
int piscadasRestantes = 0;

unsigned long efeitoTimer = 0;
int efeitoPasso = 0;

unsigned long cardioTimer = 0;
int cardioEstado = 0;
const unsigned long TEMPO_BATIDA_1 = 100;
const unsigned long TEMPO_PAUSA_CURTA = 200;
const unsigned long TEMPO_BATIDA_2 = 100;
const unsigned long TEMPO_PAUSA_LONGA = 600;

unsigned long ultimoPlayMP3 = 0;
const unsigned long duracaoAudio = 14000;

unsigned long lastDebounceTime = 0;
const unsigned long debounceDelay = 50;
unsigned long lastBotaoCoracaoTime = 0;

String audioVersaoAtual = "1.0";
unsigned long ultimaVerificacaoAudio = 0;
const unsigned long INTERVALO_VERIFICACAO_AUDIO = 3600000;
bool audioPersonalizado = false;
String audioUrlAtual = "";

void salvarConfiguracao() {
  EEPROM.write(0, corAtual);
  EEPROM.write(1, brilhoAtual);
  EEPROM.write(2, audioPersonalizado ? 1 : 0);
  EEPROM.commit();
  Serial.println("💾 Configuração salva");
}

void carregarConfiguracao() {
  corAtual = EEPROM.read(0);
  brilhoAtual = EEPROM.read(1);
  audioPersonalizado = EEPROM.read(2) == 1;
  
  if (corAtual < 0 || corAtual >= NUM_CORES) corAtual = 0;
  if (brilhoAtual < BRILHO_MIN || brilhoAtual > BRILHO_MAX) brilhoAtual = BRILHO_MAX;
  
  Serial.printf("📂 Carregado: Cor %d, Brilho %d, Audio Personalizado: %s\n", 
                corAtual, brilhoAtual, audioPersonalizado ? "SIM" : "NÃO");
}

void enviarComandoMP3(byte comando, int parametro = 0) {
  byte commandData[10] = {
    0x7E, 0xFF, 0x06, comando, 0x00,
    highByte(parametro), lowByte(parametro), 0x00, 0x00, 0xEF
  };
  
  int soma = 0;
  for (int i = 1; i < 7; i++) soma += commandData[i];
  int checksum = -soma;
  commandData[7] = highByte(checksum);
  commandData[8] = lowByte(checksum);
  
  mp3Serial.write(commandData, 10);
  Serial.print("📨 Comando MP3: 0x");
  Serial.println(comando, HEX);
  delay(100);
}

void tocarMP3() {
  if (millis() - ultimoPlayMP3 < duracaoAudio) {
    Serial.println("⏳ Áudio já tocando");
    return;
  }
  
  Serial.println("🎵 Play MP3");
  enviarComandoMP3(0x03, 1);
  ultimoPlayMP3 = millis();
}

void setarVolume(int volume) {
  enviarComandoMP3(0x06, volume);
}

bool inicializarSD() {
  if (!SD_MMC.begin()) {
    Serial.println("❌ Erro ao montar SD Card");
    return false;
  }
  
  uint8_t cardType = SD_MMC.cardType();
  if (cardType == CARD_NONE) {
    Serial.println("❌ Nenhum SD Card conectado");
    return false;
  }
  
  Serial.println("✅ SD Card montado");
  return true;
}

bool baixarAudioPersonalizado(String url) {
  Serial.printf("\n📥 INICIANDO DOWNLOAD: %s\n", url.c_str());
  
  if (!SD_MMC.begin()) {
    Serial.println("❌ SD Card não disponível");
    return false;
  }

  if (SD_MMC.exists("/01_new.mp3")) {
    SD_MMC.remove("/01_new.mp3");
  }
  
  HTTPClient http;
  http.begin(url);
  http.setTimeout(30000);
  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);
  
  int httpCode = http.GET();
  Serial.printf("📥 Código HTTP do download: %d\n", httpCode);
  
  if (httpCode != 200) {
    Serial.printf("❌ Erro no download: %d\n", httpCode);
    http.end();
    return false;
  }
  
  int totalSize = http.getSize();
  if (totalSize <= 0) {
    totalSize = 1024 * 1024;
  }
  
  Serial.printf("📦 Tamanho: %d bytes\n", totalSize);
  
  File audioFile = SD_MMC.open("/01_new.mp3", FILE_WRITE);
  if (!audioFile) {
    Serial.println("❌ Erro ao criar arquivo 01.mp3 no SD");
    http.end();
    return false;
  }
  
  uint8_t buff[256] = { 0 };
  WiFiClient *stream = http.getStreamPtr();
  int bytesBaixados = 0;
  int ultimoPercentual = 0;
  
  while (http.connected() && bytesBaixados < totalSize) {
    size_t size = stream->available();
    if (size) {
      int c = stream->readBytes(buff, ((size > sizeof(buff)) ? sizeof(buff) : size));
      audioFile.write(buff, c);
      bytesBaixados += c;
      
      int percentual = (bytesBaixados * 100) / totalSize;
      if (percentual >= ultimoPercentual + 10) {
        ultimoPercentual = percentual;
        Serial.printf("⏳ Download: %d%%\n", percentual);
      }
    }
    delay(1);
  }
  
  audioFile.close();
  http.end();
  
  Serial.printf("✅ Download concluído: %d bytes\n", bytesBaixados);

  if (bytesBaixados <= 0) {
    Serial.println("❌ Download vazio, não substituindo arquivo atual");
    SD_MMC.remove("/01_new.mp3");
    return false;
  }

  if (SD_MMC.exists("/01.mp3")) {
    SD_MMC.remove("/01.mp3");
  }

  if (!SD_MMC.rename("/01_new.mp3", "/01.mp3")) {
    Serial.println("❌ Falha ao renomear 01_new.mp3 para 01.mp3");
    SD_MMC.remove("/01_new.mp3");
    return false;
  }

  Serial.println("✅ Arquivo 01.mp3 substituído com sucesso");
  return true;
}

void verificarAudioPersonalizado() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("⚠️ WiFi desconectado, não é possível verificar áudio");
    return;
  }
  
  Serial.println("\n🔍 VERIFICANDO ÁUDIO PERSONALIZADO...");
  Serial.printf("📌 Totem ID: %s\n", DEVICE_ID);
  Serial.printf("🌐 Servidor: %s\n", SERVER_URL);
  
  HTTPClient http;
  String url = String(SERVER_URL) + "/api/audio/" + DEVICE_ID;
  Serial.printf("📡 URL: %s\n", url.c_str());
  
  http.begin(url);
  http.setTimeout(10000);
  
  int httpCode = http.GET();
  Serial.printf("📥 Código HTTP: %d\n", httpCode);
  
  if (httpCode == 200) {
    String payload = http.getString();
    Serial.printf("📦 Resposta: %s\n", payload.c_str());
    
    DynamicJsonDocument doc(1024);
    DeserializationError error = deserializeJson(doc, payload);
    
    if (!error) {
      const char* urlAudio = doc["url"];
      const char* versao = doc["versao"];
      
      if (urlAudio && strlen(urlAudio) > 0) {
        String novaUrl = String(urlAudio);
        String novaVersao = String(versao);
        
        Serial.printf("🎵 URL do áudio: %s\n", novaUrl.c_str());
        Serial.printf("🔢 Versão: %s\n", novaVersao.c_str());
        Serial.printf("📊 Versão atual: %s\n", audioVersaoAtual.c_str());
        
        if (novaVersao != audioVersaoAtual) {
          Serial.println("🆕 NOVA VERSÃO DETECTADA! Baixando...");
          
          if (baixarAudioPersonalizado(novaUrl)) {
            audioVersaoAtual = novaVersao;
            audioPersonalizado = true;
            audioUrlAtual = novaUrl;
            salvarConfiguracao();
            
            Serial.println("✅ Áudio atualizado com sucesso!");
            
            delay(1000);
            tocarMP3();
          } else {
            Serial.println("❌ Falha no download do áudio");
          }
        } else {
          Serial.println("✓ Áudio já está na versão mais recente");
        }
      } else {
        Serial.println("ℹ️ Nenhum áudio personalizado encontrado");
        audioPersonalizado = false;
      }
    } else {
      Serial.printf("❌ Erro ao parsear JSON: %s\n", error.c_str());
    }
  } else if (httpCode == 404) {
    Serial.println("❌ Totem não encontrado no servidor");
  } else {
    Serial.printf("❌ Erro HTTP: %d\n", httpCode);
    Serial.printf("🔧 Resposta: %s\n", http.getString().c_str());
  }
  
  http.end();
  Serial.println("🔍 FIM DA VERIFICAÇÃO\n");
}

void atualizarLedQrCode() {
  if (!ledQrCodeAtivo) {
    digitalWrite(LED_PIN, LOW);
    return;
  }
  
  if (millis() - ledQrCodeTimer >= 200) {
    ledQrCodeTimer = millis();
    ledQrCodeState = !ledQrCodeState;
    digitalWrite(LED_PIN, ledQrCodeState);
    
    if (!ledQrCodeState) {
      piscadasRestantes--;
      if (piscadasRestantes <= 0) {
        ledQrCodeAtivo = false;
        digitalWrite(LED_PIN, LOW);
      }
    }
  }
}

void ativarLedQrCode() {
  ledQrCodeAtivo = true;
  piscadasRestantes = ledQrCodePiscadas * 2;
  ledQrCodeState = false;
  ledQrCodeTimer = millis();
}

void atualizarBatimentoCardiaco() {
  unsigned long now = millis();
  
  switch(cardioEstado) {
    case 0:
      if (now - cardioTimer >= TEMPO_PAUSA_LONGA) {
        cardioTimer = now;
        cardioEstado = 1;
        fitaCardio.fill(0xFF0000, 0, NUM_LEDS_CARDIO);
        fitaCardio.show();
      } else {
        fitaCardio.clear();
        fitaCardio.show();
      }
      break;
    case 1:
      if (now - cardioTimer >= TEMPO_BATIDA_1) {
        cardioTimer = now;
        cardioEstado = 2;
        fitaCardio.clear();
        fitaCardio.show();
      }
      break;
    case 2:
      if (now - cardioTimer >= TEMPO_PAUSA_CURTA) {
        cardioTimer = now;
        cardioEstado = 3;
        fitaCardio.fill(0xB20000, 0, NUM_LEDS_CARDIO);
        fitaCardio.show();
      }
      break;
    case 3:
      if (now - cardioTimer >= TEMPO_BATIDA_2) {
        cardioTimer = now;
        cardioEstado = 0;
        fitaCardio.clear();
        fitaCardio.show();
      }
      break;
  }
}

void efeitoArcoIrisRapido() {
  if (millis() - efeitoTimer > 15) {
    efeitoTimer = millis();
    efeitoPasso = (efeitoPasso + 5) % 360;
    
    for (int i = 0; i < NUM_LEDS; i++) {
      int hue = (i * 6 + efeitoPasso) % 360;
      fitaPrincipal.setPixelColor(i, fitaPrincipal.ColorHSV(hue * 182, 255, brilhoAtual));
    }
    fitaPrincipal.show();
  }
}

void aplicarModoEstatico() {
  if (efeitoAtivo) return;
  
  if (corAtual < 10) {
    uint32_t cor = cores[corAtual];
    fitaPrincipal.fill(cor, 0, NUM_LEDS);
    fitaPrincipal.setBrightness(brilhoAtual);
    fitaPrincipal.show();
  }
  else if (corAtual == 10) {
    static unsigned long lastUpdate = 0;
    static int passo = 0;
    if (millis() - lastUpdate > 100) {
      lastUpdate = millis();
      passo = (passo + 1) % 360;
      for (int i = 0; i < NUM_LEDS; i++) {
        int hue = (i * 2 + passo) % 360;
        fitaPrincipal.setPixelColor(i, fitaPrincipal.ColorHSV(hue * 182, 255, brilhoAtual));
      }
      fitaPrincipal.show();
    }
  }
  else {
    fitaPrincipal.clear();
    fitaPrincipal.show();
  }
}

void executarAcao() {
  Serial.println("🎉 QR Code! Iniciando...");
  ativarLedQrCode();
  tocarMP3();
  
  efeitoAtivo = true;
  efeitoInicio = millis();
  efeitoTimer = millis();
  efeitoPasso = 0;
}

void mudarCor() {
  corAtual = (corAtual + 1) % NUM_CORES;
  Serial.printf("🎨 Modo: %d\n", corAtual);
  if (!efeitoAtivo) aplicarModoEstatico();
  salvarConfiguracao();
}

void aumentarBrilho() {
  brilhoAtual = min(brilhoAtual + 20, BRILHO_MAX);
  Serial.printf("☀️ Brilho: %d\n", brilhoAtual);
  if (!efeitoAtivo && corAtual < 10) aplicarModoEstatico();
  salvarConfiguracao();
}

void diminuirBrilho() {
  brilhoAtual = max(brilhoAtual - 20, BRILHO_MIN);
  Serial.printf("🌙 Brilho: %d\n", brilhoAtual);
  if (!efeitoAtivo && corAtual < 10) aplicarModoEstatico();
  salvarConfiguracao();
}

void callback(char* topic, byte* payload, unsigned int length) {
  String msg;
  for (int i = 0; i < length; i++) msg += (char)payload[i];
  
  String topicoStr = String(topic);
  
  if (msg == "play") {
    executarAcao();
  }
  else if (topicoStr.endsWith("/audio")) {
    Serial.println("🎵 Notificação: Novo áudio!");
    delay(1000);
    verificarAudioPersonalizado();
  } else if (msg == "atualizar_audio") {
    Serial.println("🎵 Notificação: Atualizar áudio!");
    delay(1000);
    verificarAudioPersonalizado();
  }
}

void conectarMQTT() {
  while (!client.connected()) {
    String clientId = "TOTEM-" + String(DEVICE_ID) + "-" + String(random(0xffff), HEX);
    if (client.connect(clientId.c_str())) {
      client.subscribe((String("totem/") + DEVICE_ID).c_str());
      client.subscribe((String("totem/") + DEVICE_ID + "/audio").c_str());
      Serial.println("✅ MQTT conectado!");
    } else {
      Serial.println("❌ Falha MQTT, tentando em 2s");
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  randomSeed(analogRead(0));
  EEPROM.begin(EEPROM_SIZE);
  
  pinMode(RESET_BUTTON, INPUT_PULLUP);
  pinMode(BOTAO_COR, INPUT_PULLUP);
  pinMode(BOTAO_MAIS, INPUT_PULLUP);
  pinMode(BOTAO_MENOS, INPUT_PULLUP);
  pinMode(BOTAO_CORACAO, INPUT_PULLUP);
  pinMode(LED_PIN, OUTPUT);
  
  fitaPrincipal.begin();
  fitaPrincipal.show();
  fitaCardio.begin();
  fitaCardio.show();
  
  mp3Serial.begin(9600, SERIAL_8N1, MP3_RX_PIN, MP3_TX_PIN);
  delay(1000);
  
  Serial.println("\n\n🚀 TOTEM v4.0 - SERVIDOR NUVEM");
  Serial.println("==================================");
  
  carregarConfiguracao();
  
  if (inicializarSD()) {
    Serial.println("✅ SD Card pronto");
  }
  
  if (digitalRead(RESET_BUTTON) == LOW) {
    delay(5000);
    if (digitalRead(RESET_BUTTON) == LOW) {
      wm.resetSettings();
      ESP.restart();
    }
  }
  
  wm.autoConnect("TOTEM_SETUP", "12345678");
  Serial.println("✅ WiFi conectado!");
  
  client.setServer("broker.hivemq.com", 1883);
  client.setCallback(callback);
  
  setarVolume(20);
  
  verificarAudioPersonalizado();
  
  aplicarModoEstatico();
  cardioTimer = millis();
  
  Serial.println("\n✅ Totem pronto!");
  Serial.printf("📌 ID: %s\n", DEVICE_ID);
  Serial.printf("🌐 Servidor: %s\n", SERVER_URL);
  Serial.printf("🎵 Áudio personalizado: %s\n", audioPersonalizado ? "SIM" : "NÃO");
  Serial.println("==================================\n");
}

void loop() {
  static unsigned long resetPressTime = 0;
  static bool resetPressing = false;
  
  if (digitalRead(RESET_BUTTON) == LOW) {
    if (!resetPressing) {
      resetPressing = true;
      resetPressTime = millis();
    }
    if (millis() - resetPressTime > 5000) {
      wm.resetSettings();
      ESP.restart();
    }
  } else {
    resetPressing = false;
  }
  
  if (digitalRead(BOTAO_COR) == LOW && millis() - lastDebounceTime > debounceDelay) {
    mudarCor();
    lastDebounceTime = millis();
    delay(200);
  }
  if (digitalRead(BOTAO_MAIS) == LOW && millis() - lastDebounceTime > debounceDelay) {
    aumentarBrilho();
    lastDebounceTime = millis();
    delay(200);
  }
  if (digitalRead(BOTAO_MENOS) == LOW && millis() - lastDebounceTime > debounceDelay) {
    diminuirBrilho();
    lastDebounceTime = millis();
    delay(200);
  }
  
  if (digitalRead(BOTAO_CORACAO) == LOW) {
    if (millis() - lastBotaoCoracaoTime > debounceDelay) {
      lastBotaoCoracaoTime = millis();
      delay(50);
      if (digitalRead(BOTAO_CORACAO) == LOW) {
        executarAcao();
      }
    }
  }
  
  if (efeitoAtivo) {
    if (millis() - efeitoInicio < duracaoAudio) {
      efeitoArcoIrisRapido();
    } else {
      efeitoAtivo = false;
      aplicarModoEstatico();
    }
  } else {
    aplicarModoEstatico();
  }
  
  atualizarLedQrCode();
  atualizarBatimentoCardiaco();
  
  if (!client.connected()) {
    conectarMQTT();
  }
  client.loop();
  
  if (millis() - ultimaVerificacaoAudio > INTERVALO_VERIFICACAO_AUDIO) {
    ultimaVerificacaoAudio = millis();
    verificarAudioPersonalizado();
  }
}
